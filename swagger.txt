@Bean
	public Docket productApi() {
		return new Docket(DocumentationType.SWAGGER_2)
				.select()
				.apis(RequestHandlerSelectors.any())
		//		.apis(RequestHandlerSelectors.basePackage("com.trg"))
				.paths(PathSelectors.any())
		//		.paths(PathSelectors.ant("/employees"))
				.build()
				.apiInfo(metaInfo());
		
	}
	
	private ApiInfo metaInfo() {

        ApiInfo apiInfo = new ApiInfo(
        		"Sping Boot Employee Application",
        		"Developed during Spring Boot Rest Training in November 2021",
        		"1.0",
        		"",
        		"Ramana Reddy",
        		"Licence 2.1.0",
        		"https://www.trg.com"
        );

        return apiInfo;
    }
